Create DAX Measures 12

Steps: 

1.     Go to “Report view” in Power BI.

2.     Right-click on the "US_Perm_Visas" table in the Fields pane.

3.     Select "New Measure" to create a new measure.

4.     Use the provided DAX formulas to create the desired KPIs, sub-KPIs and Arrow and save them.

5.     Repeat the above steps for each of the desired KPIs, sub-KPIs and Arrow.

 

 DAX Formulas: 

1.     Applications = 

COUNT('US_Perm_Visas'[Application_Type])

 

2.     Total Applications =

COUNTROWS(US_Perm_Visas)

 

3.     Applications per Mth = OUNT('US_Perm_Visas'[Application_Type])

 

4.     Total App This Yr =

 CALCULATE(COUNTROWS(US_Perm_Visas),     YEAR(US_Perm_Visas[Decision_Date]) = YEAR(TODAY()))

 

5.     AR LY = 

CALCULATE('US_Perm_Visas'[.Approval Rating],'US_Perm_Visas'[YEAR] = MAX('US_Perm_Visas'[YEAR]) - 1)

 

6.     Total App This Yr =

 CALCULATE(COUNTROWS(US_Perm_Visas), YEAR(US_Perm_Visas[Decision_Date]) = YEAR(TODAY()))

 

7.     Total App Last Yr = 

CALCULATE(COUNTROWS(US_Perm_Visas), YEAR(US_Perm_Visas[Decision_Date]) = YEAR(TODAY()) - 1)

 

8.     AR This Yr = CALCULATE( CALCULATE('US_Perm_Visas'[.Approval Rating],'US_Perm_Visas'[YEAR] = MAX('US_Perm_Visas'[YEAR])))

 

9.     AR YoY % = 

VAR ARThisYr = [.AR This Yr]

VAR ARLY = [.AR LY] RETURN

IF( ISBLANK(ARLY), 0,

DIVIDE(ARThisYr - ARLY, ARLY, 0))

 

10.  Visa Applications per Yr = 

 VAR VisaCount = COUNTROWS('US_Perm_Visas')

 RETURN IF( ISBLANK(VisaCount),

 BLANK(),  IF(HASONEVALUE('Calendar'[Year]),

 VisaCount, BLANK() ) )

 

11. App Per Yr = 

VAR YearCounts =  SUMMARIZE('US_Perm_Visas', US_Perm_Visas'[Year], "YearCount", COUNT('US_Perm_Visas'[Application_Type]))

VAR TotalYears = COUNTROWS(YearCounts)

VAR TotalApplications = SUMX(YearCounts, [YearCount])

RETURN DIVIDE(TotalApplications, TotalYears, 0)

 

12. Applications per Mth YoY % = 

VAR CurrentYearCount = [.Applications per Mth]

VAR PreviousYearCount = CALCULATE(

[.Applications per Mth], SAMEPERIODLASTYEAR('Calendar'[Date])

)RETURN IF(OR(ISBLANK(CurrentYearCount), ISBLANK(PreviousYearCount)),

0, DIVIDE(CurrentYearCount - PreviousYearCount, PreviousYearCount, 0))

 

13. Approval Rating = 

IF(ISBLANK([.Total Applications]) || ISBLANK(CALCULATE([.Total Applications], US_Perm_Visas[Case_Status] = "Certified-Expired")),0,DIVIDE(CALCULATE([.Total Applications], US_Perm_Visas[Case_Status] = "Certified-Expired"),

[.Total Applications]))

 

14.  Approved Applications = 

IF(ISBLANK(CALCULATE(COUNT('US_Perm_Visas'[Application_Type]),  'US_Perm_Visas'[Case_Status] = "Certified")), 0, -- Return 0 instead of BLANK for missing values CALCULATE( COUNT('US_Perm_Visas'[Application_Type]), 'US_Perm_Visas'[Case_Status] = "Certified"))

 

15. Apps YoY % = 

VAR SelectedYears = VALUES('Calendar'[Date].[Year])

RETURN  AVERAGEX ( SelectedYears, VAR TotalAppThisYr = COUNTROWS('US_Perm_Visas') VAR TotalAppLY = CALCULATE(

COUNTROWS('US_Perm_Visas'), SAMEPERIODLASTYEAR('Calendar'[Date]))

RETURN IF(ISBLANK(TotalAppThisYr) || ISBLANK(TotalAppLY),

0, -- Return 0 instead of BLANK for missing values IF(

TotalAppLY = 0, 0, -- Return 0 instead of BLANK for division by   zero DIVIDE(TotalAppThisYr - TotalAppLY, TotalAppLY))))

 

16. Wage = 

VAR TotalSalary = SUMX(US_Perm_Visas,

IF(NOT(ISBLANK(US_Perm_Visas[Wage_Offer])) && US_Perm_Visas[Wage_Offer_Unit] = "yr",

US_Perm_Visas[Wage_Offer] / 2080,  -- Yearly salary converted to hourly rate  US_Perm_Visas[Wage_Offer] -- Hourly rate ) )

VAR TotalHourlyWages = CALCULATE(SUMX( US_Perm_Visas, IF(NOT(ISBLANK(US_Perm_Visas[Wage_Offer])) && US_Perm_Visas[Wage_Offer_Unit] = "hr",  US_Perm_Visas[Wage_Offer], 0 ))) VAR CountHourlyWages = CALCULATE(COUNT(US_Perm_Visas[Wage_Offer]),

NOT(ISBLANK(US_Perm_Visas[Wage_Offer])) && US_Perm_Visas[Wage_Offer_Unit] = "hr" ) VAR TotalWages = TotalSalary + TotalHourlyWages VAR CountWages =

COUNT(US_Perm_Visas[Wage_Offer]) VAR WageMultiplier =  1.2  -- Adjust this multiplier as needed to increase wages 

RETURN  DIVIDE(TotalWages, CountWages) * WageMultiplier

 

17. Wage LY = 

CALCULATE( SUM('US_Perm_Visas'[Wage_Offer]),

YEAR('US_Perm_Visas'[Decision_Date]) = YEAR(TODAY()) - 1)

 

18. Wage This Yr = 

CALCULATE( SUM('US_Perm_Visas'[Wage_Offer]),

YEAR('US_Perm_Visas'[Decision_Date]) = YEAR(TODAY()))

 

19. Wage YoY % = 

VAR SelectedYears = VALUES('Calendar'[Date].[Year]) RETURN  AVERAGEX ( SelectedYears, VAR WageThisYr = ALCULATE(SUM('US_Perm_Visas'[Wage_Offer]), 'Calendar'[Date].[Year] = EARLIER('Calendar'[Date].[Year])) VAR WageLY = ALCULATE(SUM('US_Perm_Visas'[Wage_Offer]), 'Calendar'[Date].[Year] = EARLIER('Calendar'[Date].[Year]) - 1) VAR WageYoY = IF(ISBLANK(WageLY), 0, DIVIDE(WageThisYr -   WageLY, WageLY, 0)) RETURN IF(ISBLANK(WageYoY), 0, WageYoY) )

 

20. Applications per Mth YoY %  Arrow = 

SWITCH(TRUE(), [.Apps YoY %] <0, UNICHAR(9660), UNICHAR(9650))

 

21. AR YOY Arrow = 

SWITCH(TRUE(), [.AR YoY %] <0, UNICHAR(9660), UNICHAR(9650))

 

22. Wage YoY Arrow = 

SWITCH(TRUE(), [.Wage YoY %] <0, UNICHAR(9660), UNICHAR(9650))
